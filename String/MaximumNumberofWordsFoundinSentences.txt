class Solution {
    public int mostWordsFound(String[] sentences) {
        int max = 0;
        for(String s: sentences){
            int count = 0;
            for(char c: s.toCharArray()){
                if(c == ' '){
                    count++;
                }
            }
            max = Math.max(max, count);
        }
        return max + 1;
    }
}
