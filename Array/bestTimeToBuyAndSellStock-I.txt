class Solution {
    public int maxProfit(int[] prices) {
        int lsf = Integer.MAX_VALUE;
        int pist = 0;
        int op = 0;
        
        for(int i = 0; i < prices.length; i++){
            lsf = Math.min(lsf, prices[i]);
            
            pist = prices[i] - lsf;
            op = Math.max(pist, op);
        }
        return op;
    }
}
